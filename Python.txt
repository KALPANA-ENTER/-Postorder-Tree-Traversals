class Node:
    def __init__(self, item):
        self.left = None
        self.right = None
        self.val = item


def postorder(root):

    if root:
        # Traverse left
        postorder(root.left)
        # Traverse right
        postorder(root.right)
        # Traverse root
        print(str(root.val) + "->", end='')


root = Node(1)
root.left = Node(2)
root.right = Node(3)
root.left.left = Node(4)
root.left.right = Node(5)


print("\nPostorder traversal ")
postorder(root)
